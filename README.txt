CREATE TABLE "ArkCliente" 
   (
   "id" NUMBER NOT NULL ENABLE, 
	"cpf" VARCHAR2(14 BYTE) NOT NULL ENABLE, 
	"sexo" VARCHAR2(1 BYTE) NOT NULL ENABLE, 
	"nascimento" DATE NOT NULL ENABLE, 
	"cnpj" VARCHAR2(18 BYTE) NOT NULL ENABLE, 
	"razaoSocial" VARCHAR2(250 BYTE) NOT NULL ENABLE, 
  	"inscEstadual" VARCHAR2(15 BYTE) NOT NULL ENABLE, 
	"nome" VARCHAR2(50 BYTE) NOT NULL ENABLE, 
	"email" VARCHAR2(250 BYTE) NOT NULL ENABLE, 
	"senha" VARCHAR2(25 BYTE) NOT NULL ENABLE, 
	"telefone1" VARCHAR2(15 BYTE) NOT NULL ENABLE, 
	"telefone2" VARCHAR2(15 BYTE), 
	"idEndereco" NUMBER NOT NULL ENABLE,  
	 CONSTRAINT "ArkClientePJ_PK" PRIMARY KEY ("id"),
	 CONTRAINT "ArkClienteEnderecoFK" FOREIGN KEY ("idEndereco")
	 	REFERENCES "ArkEndereco" ("id")
	);

CREATE SEQUENCE ARKCLIENTEPJ_SEQ INCREMENT BY 1 START WITH 1 NOCACHE NOCYCLE;


CREATE TABLE "ArkEndereco"
   (
   "id" NUMBER NOT NULL ENABLE, 
	"logradouro" VARCHAR2(50 BYTE) NOT NULL ENABLE, 
	"numero" NUMBER NOT NULL ENABLE, 
	"bairro" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"cidade" VARCHAR2(50 BYTE) NOT NULL ENABLE, 
	"uf" VARCHAR2(2 BYTE) NOT NULL ENABLE, 
	"cep" VARCHAR2(10 BYTE) NOT NULL ENABLE, 
	"complemento" VARCHAR2(250 BYTE), 
	 CONSTRAINT "ArkEndereco_PK" PRIMARY KEY ("id")
	 );
	 
CREATE SEQUENCE ARKENDERECO_SEQ INCREMENT BY 1 START WITH 1 NOCACHE NOCYCLE;


CREATE TABLE "ArkProduto"
   (
   "id" NUMBER NOT NULL ENABLE,
	"nome" VARCHAR2(20) NOT NULL ENABLE,
	"preco" DECIMAL NOT NULL ENABLE,
	"descricao" VARCHAR2(250),
	"marca" VARCHAR2(20) NOT NULL ENABLE,
	"status" VARCHAR2(1) NOT NULL ENABLE,
	CONSTRAINT "ArkProduto_PK" PRIMARY KEY ("id")
	);
	
CREATE SEQUENCE ARKPRODUTO_SEQ INCREMENT BY 1 START WITH 1 NOCACHE NOCYCLE;


CREATE TABLE "SYSTEM"."ArkPedido" 
   (
    "id" NUMBER NOT NULL ENABLE, 
	"valorTotal" NUMBER(*,0) NOT NULL ENABLE, 
	"status" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	CONSTRAINT "ArkPedido_PK" PRIMARY KEY ("id")
	);
	 
CREATE SEQUENCE ARKPEDIDO_SEQ INCREMENT BY 1 START WITH 1 NOCACHE NOCYCLE;


CREATE TABLE "SYSTEM"."ArkPedidoProduto"
  ( 
    "idPedido" NUMBER NOT NULL ENABLE,
    "idProduto" NUMBER NOT NULL ENABLE,
    "quantidade" NUMBER NOT NULL ENABLE,  
    CONSTRAINT "ArkPedidoProduto_PK" PRIMARY KEY ("idPedido","idProduto"),
    CONSTRAINT "ArkPedidoFK" FOREIGN KEY ("idPedido")
     	REFERENCES "ArkPedido" ("id"),
    CONSTRAINT "ArkProdutoFK" (id) FOREIGN KEY ("idProduto")
     	REFERENCES "ArkProduto" ("id")
    );
